\newcommand{\firstname}[1]{%
  \noindent\huge\color{black}\MakeUppercase{#1}
}
%
\newcommand{\lastname}[1]{%
  \noindent\huge\color{name}\bf{\MakeUppercase{#1}}
}
%
\newcommand{\address}[1]{%
  \faMapMarker\gap\textcolor{black}{#1}
}
%
\newcommand{\email}[1]{%
  \faInbox\gap\textcolor{link}{\uline{#1}}
}
%
\newcommand{\phone}[1]{%
  \faPhone\gap\textcolor{black}{#1}
}
%
\newcommand{\linkedin}[1]{%
  % \faLinkedin%
  % \gap%
  \textcolor{link}{\uline{#1}} 
}
%
\newcommand{\github}[1]{%
  % \faGithub%
  % \gap%
  \textcolor{link}{\uline{#1}}   
}
%
\newcommand{\web}[1]{%
  \textcolor{link}{\uline{#1}}   
}

\newcommand{\gap}{\hspace{0.3em}}
%
\newcommand{\infobox}[3]{%
  \color{gray!75}\small
    \centerline{\address{#1}\gap\gap\email{#2}\gap\gap\phone{#3}%
    % \centerline{
    % \linkedin{#4}\gap\gap\github{#5}
    % }
    % \vspace{0.30em}
    }
}

\newtcolorbox{banner}[3][]{
    center,
    left=50pt,
    width=1.2\textwidth, % width to span banner
    colframe=white, 
    enhanced,
    colback={frame!15}, % background color
    attach boxed title to top center={
      yshift=-11.5mm,  % distance to shift name box from banner 
                   % neg is down, pos is up
      xshift=18pt
      },
    boxed title style={  %% border with full name
      sharp corners,
      colframe=frame, 
      enhanced, 
      colback=white,
      left=40pt,
      right=35pt, % requires some offset from left due to font
      top=0.3em,  % top of name to top of name box
      bottom=1.5em, % bottom of name to bottom of name box
      halign=center,
      boxrule=2pt,
    },
    top=3.5mm,
    title=\firstname{#2}\ \lastname{#3},#1
}
