% hspace in marginsym & school tcolorbox 
\newcommand{\marginsym}[1]{%
  {\normalsize\textcolor{ltgray}{#1}\par}
}
%
\newcommand{\margintable}[1]{%
  \begin{tcolorbox}[
  enhanced,
  colframe=white, 
  colback=white,
  halign=right, 
  boxrule=0em, 
  boxsep=0em,
  left=0em,top=0.0em,
  bottom=0.0em,
  right=-17pt,
  ]
  {#1}
  \end{tcolorbox}
}
%

  \begin{tcolorbox}[
  enhanced,
  colframe=white, 
  colback=white,
  halign=right, 
  boxrule=0em, 
  boxsep=0em,
  left=0em,top=0.0em,
  bottom=0.0pt,

  {#4} & \ 
  \end{tabular}
  }
  }
